-- Profili
CREATE TABLE profiles (
  id uuid REFERENCES auth.users PRIMARY KEY,
  ristorante text NOT NULL,
  role text DEFAULT 'user',
  subscription_status text DEFAULT 'inactive',
  stripe_customer_id text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "own" ON profiles FOR ALL USING (auth.uid()=id);
CREATE POLICY "admin_all" ON profiles FOR ALL USING (
  EXISTS(SELECT 1 FROM profiles WHERE id=auth.uid() AND role='admin'));

-- Codici invito
CREATE TABLE invite_codes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  code text UNIQUE NOT NULL,
  used boolean DEFAULT false,
  used_by uuid,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE invite_codes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public_check" ON invite_codes FOR SELECT USING (true);
CREATE POLICY "admin_manage" ON invite_codes FOR ALL USING (
  EXISTS(SELECT 1 FROM profiles WHERE id=auth.uid() AND role='admin'));

-- No-show
CREATE TABLE noshow (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  phone text NOT NULL,
  name text,
  date date,
  note text,
  ristorante text,
  user_id uuid REFERENCES auth.users,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE noshow ENABLE ROW LEVEL SECURITY;
CREATE POLICY "active_read" ON noshow FOR SELECT USING (
  EXISTS(SELECT 1 FROM profiles WHERE id=auth.uid() AND subscription_status='active'));
CREATE POLICY "active_insert" ON noshow FOR INSERT WITH CHECK (
  EXISTS(SELECT 1 FROM profiles WHERE id=auth.uid() AND subscription_status='active'));
